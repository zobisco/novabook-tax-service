// Enum definition
Enum Event_Type {
  "SALES"
  "TAX_PAYMENT"
}

Table Transaction {
  id        TEXT PK
  eventType Event_Type
  date      TEXT UNIQUE
  invoiceId TEXT
  amount    INTEGER
}

Table SaleItem {
  id            TEXT PK
  transactionId TEXT
  itemId        TEXT UNIQUE
  cost          INTEGER UNIQUE
  taxRate       REAL UNIQUE
}

Table SaleAmendment {
  id          TEXT PK
  date        TEXT UNIQUE
  invoiceId   TEXT UNIQUE
  itemId      TEXT UNIQUE
  cost        INTEGER UNIQUE
  taxRate     REAL UNIQUE
  taxPosition INTEGER
}

// Relationships
Ref: "SaleItem"."transactionId" > "Transaction"."id"
